Дочерний: >

        <li></li>
    </ul>
</nav>
Соединение: + 
<!-- div+p+bq -->
<div></div>
<p></p>
<blockquote></blockquote>
Поместить выше (в дереве HTML): ^ 
<!-- div+div>p>span+em^bq -->
<div></div>
<div>
    <p><span></span><em></em></p>
    <blockquote></blockquote>
</div>

<!-- div+div>p>span+em^^bq -->
<div></div>
<div>
    <p><span></span><em></em></p>
</div>
<blockquote></blockquote>

        </ul>
    </header>

    </footer>
